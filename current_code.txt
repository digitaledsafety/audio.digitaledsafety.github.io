                this.started = false;
                this.timeoutId = null;
                this.currentNoteIndex = 0;
                this.nextStepTime = 0;
                this.oscillators = [];
                this.gainNodes = [];
                this.midiListeners = [];
                this.tickListeners = [];
                this.tickAccumulator = 0;

                // Master gain for the Arpeggiator output
                this.mainOutput = this.audioContext.createGain();
                this.mainOutput.gain.value = 0.5; // Default volume for arpeggiator

                this.transposeControl = this.audioContext.createGain();
                this.transposeControl.gain.value = 1200; // 1V = 1200 cents

                this.setupNotesAndChords();
            }

            receiveMidiNotes(notes) {
                if (notes && notes.length > 0) {
                    const firstNote = notes[0];
                    // Find the note name from the MIDI value
                    let closestNoteName = 'C4';
                    let minDiff = Infinity;
                    for (const noteName in this.rootNotes) {
                        const diff = Math.abs(this.rootNotes[noteName] - firstNote);
                        if (diff < minDiff) {
                            minDiff = diff;
                            closestNoteName = noteName;
                        }
                    }

                    // Update the internal data and the audio output
                    this.updateParameter('rootNote', closestNoteName);

                    // Also update the UI control in the Rete node
                    const reteNode = editor.getNode(this.reteNodeId);
                    if (reteNode && reteNode.controls.rootNote) {
                        reteNode.controls.rootNote.value = closestNoteName;
                        area.update('node', this.reteNodeId); // This is important to re-render the node's control
                    }
                }
            }

            addMidiListener(callback) {
                if (typeof callback === 'function' && !this.midiListeners.includes(callback)) {
                    this.midiListeners.push(callback);
                }
            }

            removeMidiListener(callback) {
                const index = this.midiListeners.indexOf(callback);
                if (index > -1) {
                    this.midiListeners.splice(index, 1);
                }
            }

            onTick(tickTime, tickDuration) {
                const noteDurationMap = { '1': 96, '1/2': 48, '1/4': 24, '1/8': 12, '1/16': 6, '1/32': 3, '1/64': 1.5, '1/128': 0.75 };
                const ticksPerStep = noteDurationMap[this.data.noteDuration] || 6;

                this.tickAccumulator += 1;

                if (this.tickAccumulator >= ticksPerStep) {
                    this.tickAccumulator -= ticksPerStep;

                    const notes = this.getArpeggioNotes();
                    if (notes.length > 0) {
                        const currentStep = notes[this.currentNoteIndex];
                        const stepDuration = tickDuration * ticksPerStep;

                        // Schedule the note to play. Play it slightly shorter than the full step duration for a staccato effect.
                        this.scheduleNote(currentStep, stepDuration * 0.9, tickTime);

                        const midiNotesToBroadcast = Array.isArray(currentStep) ? currentStep : [currentStep];
                        if (this.midiListeners.length > 0) {
                            this.midiListeners.forEach(callback => callback(midiNotesToBroadcast));
                        }

                        this.currentNoteIndex = (this.currentNoteIndex + 1) % notes.length;
                    }
                }
            }

            setupNotesAndChords() {
                // Chord definitions: MIDI offsets from root
                this.chordTypes = CHORD_TYPES;

                // Root note MIDI numbers
                this.rootNotes = ROOT_NOTES;
            }

            getArpeggioNotes() {
                const rootMidi = this.rootNotes[this.data.rootNote] || 60; // Default to C4
                const chordOffsets = this.chordTypes[this.data.chordType] || this.chordTypes['Major Triad'];
                const octaveRange = this.data.octaveRange || 1; // Number of octaves to span
                const arpeggioNotes = [];

                for (let oct = 0; oct < octaveRange; oct++) {
                    chordOffsets.forEach(offset => {
                        arpeggioNotes.push(rootMidi + offset + (oct * 12));
                    });
                }

                // First, ensure the base set of notes is sorted in ascending order.
                const sortedNotes = [...arpeggioNotes].sort((a, b) => a - b);
                let patternResult = []; // This will hold the sequence of notes for the arpeggio

                if (this.data.arpeggioPattern === 'Custom') {
                    const indices = (this.data.pattern || '0 1 2 3').split(/\s+/).filter(s => s !== '');
                    for (const idx of indices) {
                        const i = parseInt(idx);
                        if (!isNaN(i) && i >= 0 && i < sortedNotes.length) {
                            patternResult.push(sortedNotes[i]);
                        }
                    }
                } else {
                    switch (this.data.arpeggioPattern) {
                        case 'Down':
                            patternResult = [...sortedNotes].reverse();
                            break;

                        case 'Up-Down':
                            // Plays notes up, then down without repeating the highest/lowest notes
                            const reversedUpdown = [...sortedNotes].reverse();
                            patternResult = sortedNotes.concat(reversedUpdown.slice(1, -1));
                            break;

                        case 'Down-Up':
                            const reversedDownUp = [...sortedNotes].reverse();
                            patternResult = reversedDownUp.concat(sortedNotes.slice(1, -1));
                            break;

                        case 'Converge':
                            // Starts from the lowest and highest notes, moving towards the middle
                            let lowIdxConverge = 0;
                            let highIdxConverge = sortedNotes.length - 1;
                            while (lowIdxConverge <= highIdxConverge) {
                                if (lowIdxConverge === highIdxConverge) {
                                    patternResult.push(sortedNotes[lowIdxConverge]);
                                } else {
                                    patternResult.push(sortedNotes[lowIdxConverge]);
                                    patternResult.push(sortedNotes[highIdxConverge]);
                                }
                                lowIdxConverge++;
                                highIdxConverge--;
                            }
                            break;

                        case 'Diverge':
                            // Starts from the middle notes, moving outwards towards the lowest and highest
                            const lenDiverge = sortedNotes.length;
                            let midLowIdxDiverge = Math.floor((lenDiverge - 1) / 2);
                            let midHighIdxDiverge = Math.ceil((lenDiverge - 1) / 2);

                            // If odd number of notes, the single middle note plays first
                            if (lenDiverge % 2 !== 0) {
                                patternResult.push(sortedNotes[midLowIdxDiverge]);
                                midLowIdxDiverge--;
                                midHighIdxDiverge++;
                            }

                            // Now, add notes outwards from the middle
                            while (midLowIdxDiverge >= 0 && midHighIdxDiverge < lenDiverge) {
                                patternResult.push(sortedNotes[midLowIdxDiverge]);
                                patternResult.push(sortedNotes[midHighIdxDiverge]);
                                midLowIdxDiverge--;
                                midHighIdxDiverge++;
                            }
                            break;

                        case 'Alberti Bass':
                            if (sortedNotes.length >= 3) {
                                // 1-5-3-5 pattern from the chord's notes
                                const root = sortedNotes[0];
                                const third = sortedNotes[1];
                                const fifth = sortedNotes[2];
                                patternResult = [root, fifth, third, fifth];
                            } else {
                                // Fallback for chords with less than 3 notes
                                patternResult = sortedNotes;
                            }
                            break;

                        case 'Chord':
                            // This will be a single step in the sequence, containing all notes
                            patternResult = [sortedNotes];
                            break;

                        case 'Melodic':
                            // A common "melodic" pattern is to go up, then come back down partway, or a zig-zag.
                            // Here's one interpretation: Play ascending, then descend from the second highest note down to the lowest.
                            if (sortedNotes.length === 0) {
                                patternResult = [];
                            } else if (sortedNotes.length === 1) {
                                patternResult = [...sortedNotes];
                            } else {
                                // Ascending part
                                patternResult = [...sortedNotes];

                                // Descending part from the second highest note down to the lowest
                                for (let i = sortedNotes.length - 2; i >= 0; i--) {
                                    patternResult.push(sortedNotes[i]);
                                }
                            }
                            // Example: [C, E, G, C'] -> [C, E, G, C', G, E, C]
                            break;

                        default: // 'Up' pattern
                            patternResult = sortedNotes;
                            break;
                    }
                }

                return patternResult;
            }

            scheduleNote(midiNotes, durationSeconds, startTime) {
                if (!this.audioContext) return;
                const now = startTime || this.audioContext.currentTime;

                const notesToPlay = Array.isArray(midiNotes) ? midiNotes : [midiNotes];
                notesToPlay.forEach(midiNote => {
                    const frequency = midiToFrequency(midiNote);
                    const osc = this.audioContext.createOscillator();
                    const gainNode = this.audioContext.createGain();

                    osc.type = this.data.waveform || 'sine';
                    osc.frequency.setValueAtTime(frequency, now);
                    this.transposeControl.connect(osc.detune);

                    const adsrData = {
                        attack: this.data.attack,
                        decay: this.data.decay,
                        sustain: this.data.sustain,
                        release: this.data.release
                    };
                    applyADSR(gainNode, adsrData, durationSeconds, this.audioContext, 0.3 / notesToPlay.length, now);

                    osc.connect(gainNode);
                    gainNode.connect(this.mainOutput);
                    osc.start(now);
                    osc.stop(now + durationSeconds);

                    osc.onended = () => {
                        osc.disconnect();
                        gainNode.disconnect();
                    };
                });
            }

            start() {
                if (this.started || !this.audioContext) return;
                this.started = true;
                this.currentNoteIndex = 0;
                this.nextStepTime = this.audioContext.currentTime;
                this.schedule();
            }

            schedule() {
                if (!this.started) return;

                const hasClockConnection = editor.getConnections().some(conn => conn.target === this.reteNodeId && conn.targetInput === 'clock');
                if (hasClockConnection) {
                    // In external clock mode, the onTick method will handle scheduling.
                    // This loop just needs to keep running to take over if the clock is disconnected.
                    this.timeoutId = setTimeout(() => this.schedule(), 100); // Check for clock disconnection periodically
                    return;
                }

                const notes = this.getArpeggioNotes();
                if (notes.length === 0) {
                    this.stop();
                    return;
                }

                const bpm = this.data.bpm || 120;
                const noteDurationMap = { '1': 4, '1/2': 2, '1/4': 1, '1/8': 0.5, '1/16': 0.25, '1/32': 0.125, '1/64': 0.0625, '1/128': 0.03125 };
                const noteValue = noteDurationMap[this.data.noteDuration] || 0.25;
                const secondsPerBeat = 60.0 / bpm;
                const stepDuration = secondsPerBeat * noteValue;

                while (this.nextStepTime < this.audioContext.currentTime + 0.1) {
                    const midiNote = notes[this.currentNoteIndex];
                    this.scheduleNote(midiNote, stepDuration, this.nextStepTime);

                    if (this.midiListeners.length > 0) {
                        const midiNotesToBroadcast = Array.isArray(midiNote) ? midiNote : [midiNote];
                        this.midiListeners.forEach(callback => callback(midiNotesToBroadcast));
                    }

                    this.nextStepTime += stepDuration;
                    this.currentNoteIndex = (this.currentNoteIndex + 1) % notes.length;
                }
                this.timeoutId = setTimeout(() => this.schedule(), 25);
            }

            stop() {
                if (!this.started) return;
                clearTimeout(this.timeoutId);
                this.timeoutId = null;
                this.started = false;
                console.log('Arpeggiator stopped.');
            }

            updateParameter(param, value) {
                this.data[param] = value;
                if (param === 'gain') {
                    this.mainOutput.gain.setTargetAtTime(value, this.audioContext.currentTime, 0.01);
                }
                if (this.started) {
                    // Restart arpeggiator to apply new tempo/notes/pattern
                    this.stop();
                    this.start();
                }
            }
        }


        class ArpeggiatorNode extends WebAudioNode {

                    this.controls.pattern.hidden = (val !== 'Custom');

                    const audioNode = reteAudioNodes.get(this.id);
                    if (audioNode && audioNode.updateParameter) audioNode.updateParameter('arpeggioPattern', val);
                }));

                this.data.pattern = '0 1 2 3';
                this.addControl('pattern', new TextControl('arpPatternText', 'Pattern Indices', this.data.pattern, (val) => {
                    this.data.pattern = val;
                    const audioNode = reteAudioNodes.get(this.id);
                    if (audioNode && audioNode.updateParameter) audioNode.updateParameter('pattern', val);
                }));
                this.controls.pattern.hidden = (this.data.arpeggioPattern !== 'Custom');

                this.data.octaveRange = 1;
                this.addControl('octaveRange', new SliderControl('arpOctaveRange', 'Octaves', 1, 5, 1, this.data.octaveRange, (val) => {
                    this.data.octaveRange = val;
                    const audioNode = reteAudioNodes.get(this.id);
                    if (audioNode && audioNode.updateParameter) audioNode.updateParameter('octaveRange', val);
                }));

                this.data.waveform = 'sine';
                this.addControl('waveform', new SelectControl('waveformSelect', 'Waveform', ['sine', 'square', 'sawtooth', 'triangle'], this.data.waveform, (val) => {
                    this.data.waveform = val;
                    // Arpeggiator's internal oscillators will pick this up
                    const audioNode = reteAudioNodes.get(this.id);
                    if (audioNode && audioNode.updateParameter) audioNode.updateParameter('waveform', val);
                }));

                this.data.attack = 0.01;
                this.data.decay = 0.1;
                this.data.sustain = 0.5;
                this.data.release = 0.1;

                this.addControl('envelopeToggle', new ButtonControl('envelopeToggle', 'Envelope', () => {
                    this.data.showEnvelope = !this.data.showEnvelope;
                    this.controls.attack.hidden = !this.data.showEnvelope;
                    this.controls.decay.hidden = !this.data.showEnvelope;
                    this.controls.sustain.hidden = !this.data.showEnvelope;
                    this.controls.release.hidden = !this.data.showEnvelope;
                    area.update('node', this.id);
                }));

                this.addControl('attack', new SliderControl('arpAttack', 'Attack (s)', 0.01, 2, 0.01, this.data.attack, (val) => {
                    this.data.attack = val;
                }));
                this.controls.attack.hidden = true;

                this.addControl('decay', new SliderControl('arpDecay', 'Decay (s)', 0.01, 2, 0.01, this.data.decay, (val) => {
                    this.data.decay = val;

                this.controls.decay.hidden = true;

                    this.data.sustain = val;
                }));
                this.controls.sustain.hidden = true;

                this.addControl('release', new SliderControl('arpRelease', 'Release (s)', 0.01, 5, 0.01, this.data.release, (val) => {
                    this.data.release = val;
                }));
                this.controls.release.hidden = true;
            }

            onDataLoaded() {
                this.controls.pattern.hidden = (this.data.arpeggioPattern !== 'Custom');
                const show = !!this.data.showEnvelope;
                this.controls.attack.hidden = !show;
                this.controls.decay.hidden = !show;
                this.controls.sustain.hidden = !show;
                this.controls.release.hidden = !show;
            }

            randomize() {
                const audioNode = reteAudioNodes.get(this.id);
                if (audioNode) {
                    const chordOffsets = CHORD_TYPES[this.data.chordType] || CHORD_TYPES['Major Triad'];
                    const octaveRange = this.data.octaveRange || 1;
                    const numNotes = chordOffsets.length * octaveRange;

                    let indices = [];
                    for (let i = 0; i < 16; i++) {
                        indices.push(Math.floor(Math.random() * numNotes));
                    }
                    const newPattern = indices.join(' ');
                    this.controls.pattern.value = newPattern;
                    this.data.pattern = newPattern;
                    audioNode.updateParameter('pattern', newPattern);
                    area.update('node', this.id);
                }
            }

            build() {
                return new Arpeggiator(audioContext, this.id, this.data);
            }

            start() {
                const audioNode = reteAudioNodes.get(this.id);
                if (audioNode && audioNode.start && !audioNode.started) {
                    try {
                        audioNode.start();
                        audioNode.started = true;
                    } catch (e) {
                        console.warn(`Could not start ${this.label} for node ${this.id}:`, e);
                    }
                }
            }
        }



        // --- New Drum Machine Node ---
        class DrumMachine {
            constructor(audioContext, reteNodeId, data) {
                this.audioContext = audioContext;
                this.reteNodeId = reteNodeId;
                this.data = data;
                this.started = false;
                this.currentStep = 0;
                this.sequenceEvents = [];
                this.timeoutId = null;
                this.nextStepTime = 0;
                this.midiListeners = []; // Kept for consistency, might not be used
                this.tickAccumulator = 0;

                this.mainOutput = this.audioContext.createGain();
                this.mainOutput.gain.value = 0.8; // Drums can be loud

                // Note map for drum sounds
                const drumNoteMap = { 'k': 60, 's': 62, 'h': 64 }; // Using midi numbers as placeholders
                this.parser = new MiniNotationParser(drumNoteMap);
                this.sequenceEvents = this.parseSequence();
            }

            addMidiListener(callback) {
                // Not typically used for a drum machine, but here for API consistency
            }

            removeMidiListener(callback) {
                // Not typically used for a drum machine
            }

            onTick(tickTime, tickDuration) {
                const noteDurationMap = { '1': 96, '1/2': 48, '1/4': 24, '1/8': 12, '1/16': 6, '1/32': 3, '1/64': 1.5, '1/128': 0.75 };
                const baseTicks = noteDurationMap[this.data.noteDuration] || 6;
                const event = this.sequenceEvents[this.currentStep];
                const representativeEvent = Array.isArray(event) ? event[0] : event;

                if (representativeEvent) {
                    const ticksForThisStep = baseTicks * (1 / representativeEvent.speed) * representativeEvent.duration;
                    this.tickAccumulator += 1;

                    if (this.tickAccumulator >= ticksForThisStep) {
                        this.tickAccumulator -= ticksForThisStep;

                        const stepDuration = tickDuration * ticksForThisStep;
                        if (Array.isArray(event)) {
                            for (const note of event) {
                                if (note.noteName) this.scheduleNote(note.noteName, stepDuration * note.elongation, tickTime);
                            }
                        } else {
                            if (event.noteName) this.scheduleNote(event.noteName, stepDuration * event.elongation, tickTime);
                        }
                        this.currentStep = (this.currentStep + 1) % this.sequenceEvents.length;
                        if (this.currentStep === 0) {
                            this.sequenceEvents = this.parseSequence();
                        }
                    }
                }
            }

            parseSequence() {
                return this.parser.parse(this.data.sequence || '');
            }

            // --- Sound Synthesis ---
            createKick(now) {
                const osc = this.audioContext.createOscillator();
                const gain = this.audioContext.createGain();
                osc.connect(gain);
                gain.connect(this.mainOutput);

                osc.frequency.setValueAtTime(150, now);
                osc.frequency.exponentialRampToValueAtTime(50, now + 0.1);
                gain.gain.setValueAtTime(1, now);
                gain.gain.exponentialRampToValueAtTime(0.001, now + 0.5);

                osc.start(now);
                osc.stop(now + 0.5);
            }

            createSnare(now) {
                const noise = this.audioContext.createBufferSource();
                const noiseBuffer = this.audioContext.createBuffer(1, this.audioContext.sampleRate * 0.2, this.audioContext.sampleRate);
                const output = noiseBuffer.getChannelData(0);
                for (let i = 0; i < noiseBuffer.length; i++) {
                    output[i] = Math.random() * 2 - 1;
                }
                noise.buffer = noiseBuffer;

                const noiseFilter = this.audioContext.createBiquadFilter();
                noiseFilter.type = 'highpass';
                noiseFilter.frequency.value = 1500;
                noise.connect(noiseFilter);

                const noiseEnvelope = this.audioContext.createGain();
                noiseFilter.connect(noiseEnvelope);
                noiseEnvelope.connect(this.mainOutput);

                noiseEnvelope.gain.setValueAtTime(1, now);
                noiseEnvelope.gain.exponentialRampToValueAtTime(0.01, now + 0.1);
                noise.start(now);
                noise.stop(now + 0.1);

                // Snare body
                const osc = this.audioContext.createOscillator();
                const oscEnvelope = this.audioContext.createGain();
                osc.type = 'triangle';
                osc.frequency.value = 180;
                osc.connect(oscEnvelope);
                oscEnvelope.connect(this.mainOutput);
                oscEnvelope.gain.setValueAtTime(0.5, now);
                oscEnvelope.gain.exponentialRampToValueAtTime(0.01, now + 0.08);
                osc.start(now);
                osc.stop(now + 0.08);
            }

            createHiHat(now) {
                const noise = this.audioContext.createBufferSource();
                const noiseBuffer = this.audioContext.createBuffer(1, this.audioContext.sampleRate * 0.1, this.audioContext.sampleRate);
                const output = noiseBuffer.getChannelData(0);
                for (let i = 0; i < noiseBuffer.length; i++) {
                    output[i] = Math.random() * 2 - 1;
                }
                noise.buffer = noiseBuffer;

                const noiseFilter = this.audioContext.createBiquadFilter();
                noiseFilter.type = 'highpass';
                noiseFilter.frequency.value = 7000;
                noise.connect(noiseFilter);

                const noiseEnvelope = this.audioContext.createGain();
                noiseFilter.connect(noiseEnvelope);
                noiseEnvelope.connect(this.mainOutput);
                noiseEnvelope.gain.setValueAtTime(1, now);
                noiseEnvelope.gain.exponentialRampToValueAtTime(0.01, now + 0.05);
                noise.start(now);
                noise.stop(now + 0.05);
            }


            scheduleNote(noteName, durationSeconds, startTime) {
                if (!this.audioContext) return;
                const now = startTime || this.audioContext.currentTime;

                const normalizedNote = typeof noteName === 'string' ? noteName.toLowerCase() : noteName;
                switch (normalizedNote) {
                    case 'k':
                        this.createKick(now);
                        break;
                    case 's':
                        this.createSnare(now);
                        break;
                    case 'h':
                        this.createHiHat(now);
                        break;
                }
            }

            start() {
                if (this.started || !this.audioContext) return;
                this.started = true;
                this.currentStep = 0;
                this.tickAccumulator = 999999; // Ensure immediate trigger on next tick
                this.sequenceEvents = this.parseSequence();
                this.nextStepTime = this.audioContext.currentTime;
                this.schedule();
            }

            schedule() {
                if (!this.started) return;

                const hasClockConnection = editor.getConnections().some(conn => conn.target === this.reteNodeId && conn.targetInput === 'clock');
                if (hasClockConnection) {
                    this.timeoutId = setTimeout(() => this.schedule(), 100);
                    return;
                }
                 if (!this.sequenceEvents || this.sequenceEvents.length === 0) {
                    this.stop();
                    return;
                }

                while (this.nextStepTime < this.audioContext.currentTime + 0.1) {
                    const event = this.sequenceEvents[this.currentStep];
                    const representativeEvent = Array.isArray(event) ? event[0] : event;

                    if (!representativeEvent) {
                        this.currentStep = (this.currentStep + 1) % this.sequenceEvents.length;
                        continue;
                    }

                    const bpm = this.data.bpm || 120;
                    const noteDurationMap = { '1': 1, '1/2': 0.5, '1/4': 0.25, '1/8': 0.125, '1/16': 0.0625, '1/32': 0.03125, '1/64': 0.015625, '1/128': 0.0078125 };
                    const baseNoteDuration = noteDurationMap[this.data.noteDuration] || 0.25;
                    const secondsPerBeat = 60.0 / bpm;
                    const stepDuration = secondsPerBeat * 4 * baseNoteDuration * (1 / representativeEvent.speed) * representativeEvent.duration;

                    if (Array.isArray(event)) {
                        for (const note of event) {
                            if (note.noteName) this.scheduleNote(note.noteName, stepDuration * note.elongation, this.nextStepTime);
                        }
                    } else {
                        if (event.noteName) this.scheduleNote(event.noteName, stepDuration * event.elongation, this.nextStepTime);
                    }

                    this.nextStepTime += stepDuration;
                    this.currentStep = (this.currentStep + 1) % this.sequenceEvents.length;
                }

                this.timeoutId = setTimeout(() => this.schedule(), 25);
            }

            stop() {
                if (!this.started) return;
                clearTimeout(this.timeoutId);
                this.timeoutId = null;
                this.started = false;
            }

            updateParameter(param, value) {
                this.data[param] = value;
                if (this.started) {
                    this.stop();
                    this.start();
                }
            }

            generateRandomSequence() {
                const steps = 16;
                let sequence = new Array(steps).fill('~');

                // Kick: 4-on-the-floor
                for (let i = 0; i < steps; i += 4) {
                    sequence[i] = 'k';
                }

                // Snare: steps 4 and 12
                if (sequence[4] === '~') sequence[4] = 's';
                else sequence[4] = '[' + sequence[4] + ' s]';

                if (sequence[12] === '~') sequence[12] = 's';
                else sequence[12] = '[' + sequence[12] + ' s]';

                // Random Hi-hats and occasional extra kicks/snares
                for (let i = 0; i < steps; i++) {
                    if (Math.random() < 0.5) { // 50% chance for hi-hats
                        if (sequence[i] === '~') {
                            sequence[i] = 'h';
                        } else {
                            sequence[i] = '[' + sequence[i] + ' h]';
                        }
                    }
                    if (Math.random() < 0.1 && sequence[i] === '~') {
                        sequence[i] = Math.random() < 0.5 ? 'k' : 's';
                    }
                }
                return sequence.join(' ');
            }
        }


        // --- New Sequencer Node ---
        // Internal class to manage the Web Audio API aspects of the Sequencer
        class Sequencer {
            constructor(audioContext, reteNodeId, data) {
                this.audioContext = audioContext;
                this.reteNodeId = reteNodeId;
                this.data = data; // Reference to the Rete.js node's data
                this.started = false;
                this.timeoutId = null;
                this.currentStep = 0;
                this.nextStepTime = 0;
                this.tickAccumulator = 0;
                this.sequenceEvents = null;
                this.midiListeners = [];

                // Master gain for the Sequencer output
                this.mainOutput = this.audioContext.createGain();
                this.mainOutput.gain.value = 0.5; // Default volume for sequencer

                this.transposeControl = this.audioContext.createGain();
                this.transposeControl.gain.value = 1200; // 1V = 1200 cents

                this.setupNotes();
                this.parser = new MiniNotationParser(this.rootNotes);
            }

            addMidiListener(callback) {
                if (typeof callback === 'function' && !this.midiListeners.includes(callback)) {
                    this.midiListeners.push(callback);
                }
            }

            removeMidiListener(callback) {
                const index = this.midiListeners.indexOf(callback);
                if (index > -1) {
                    this.midiListeners.splice(index, 1);
                }
            }

            onTick(tickTime, tickDuration) {
                const noteDurationMap = { '1': 96, '1/2': 48, '1/4': 24, '1/8': 12, '1/16': 6, '1/32': 3, '1/64': 1.5, '1/128': 0.75 };
                const baseTicks = noteDurationMap[this.data.noteDuration] || 6;

                if (!this.sequenceEvents) {
                    this.sequenceEvents = this.parseSequence();
                }
                if (!this.sequenceEvents || this.sequenceEvents.length === 0) return;

                const event = this.sequenceEvents[this.currentStep];
                const representativeEvent = Array.isArray(event) ? event[0] : event;

                if (representativeEvent) {
                    const ticksForThisStep = baseTicks * (1 / representativeEvent.speed) * representativeEvent.duration;
                    this.tickAccumulator += 1;

                    if (this.tickAccumulator >= ticksForThisStep) {
                        this.tickAccumulator -= ticksForThisStep;

                        const stepDuration = tickDuration * ticksForThisStep;
                        const midiNotesToBroadcast = [];

                        if (Array.isArray(event)) {
                            for (const note of event) {
                                if (note.midi !== null) {
                                    this.scheduleNote(note.midi, stepDuration * note.elongation, tickTime);
                                    midiNotesToBroadcast.push(note.midi);
                                }
                            }
                        } else {
                            if (event.midi !== null) {
                                this.scheduleNote(event.midi, stepDuration * event.elongation, tickTime);
                                midiNotesToBroadcast.push(event.midi);
                            }
                        }

                        if (this.midiListeners.length > 0 && midiNotesToBroadcast.length > 0) {
                            this.midiListeners.forEach(callback => callback(midiNotesToBroadcast));
                        }

                        this.currentStep = (this.currentStep + 1) % this.sequenceEvents.length;
                        if (this.currentStep === 0) {
                            this.sequenceEvents = this.parseSequence();
                        }
                    }
                }
            }

            setupNotes() {
                // Root note MIDI numbers
                this.rootNotes = ROOT_NOTES;
            }

            parseSequence() {
                return this.parser.parse(this.data.sequence || '');
            }
            scheduleNote(midiNote, durationSeconds, startTime) {
                if (!this.audioContext) return;
                const now = startTime || this.audioContext.currentTime;

                const frequency = midiToFrequency(midiNote);
                const osc = this.audioContext.createOscillator();
                const gainNode = this.audioContext.createGain();

                osc.type = this.data.waveform || 'sine';
                osc.frequency.setValueAtTime(frequency, now);
                this.transposeControl.connect(osc.detune);

                const adsrData = {
                    attack: this.data.attack,
                    decay: this.data.decay,
                    sustain: this.data.sustain,
                    release: this.data.release
                };
                applyADSR(gainNode, adsrData, durationSeconds, this.audioContext, 0.3, now);

                osc.connect(gainNode);
                gainNode.connect(this.mainOutput);

                osc.start(now);
                osc.stop(now + durationSeconds);

                osc.onended = () => {
                    osc.disconnect();
                    gainNode.disconnect();
                };
            }

            start() {
                if (this.started || !this.audioContext) return;
                this.started = true;
                this.currentStep = 0;
                this.sequenceEvents = this.parseSequence();
                this.nextStepTime = this.audioContext.currentTime;
                this.schedule();
            }

            schedule() {
                if (!this.started) return;

                const hasClockConnection = editor.getConnections().some(conn => conn.target === this.reteNodeId && conn.targetInput === 'clock');
                if (hasClockConnection) {
                    // In external clock mode, the onTick method will handle scheduling.
                    // This loop just needs to keep running to take over if the clock is disconnected.
                    this.timeoutId = setTimeout(() => this.schedule(), 100); // Check for clock disconnection periodically
                    return;
                }

                if (!this.sequenceEvents || this.sequenceEvents.length === 0) {
                    this.stop();
                    return;
                }

                while (this.nextStepTime < this.audioContext.currentTime + 0.1) {
                    const event = this.sequenceEvents[this.currentStep];
                    const representativeEvent = Array.isArray(event) ? event[0] : event;

                    if (!representativeEvent) {
                        this.currentStep = (this.currentStep + 1) % this.sequenceEvents.length;
                        continue; // Skip to the next iteration of the while loop
                    }

                    const bpm = this.data.bpm || 120;
                    const noteDurationMap = { '1': 1, '1/2': 0.5, '1/4': 0.25, '1/8': 0.125, '1/16': 0.0625, '1/32': 0.03125, '1/64': 0.015625, '1/128': 0.0078125 };
                    const baseNoteDuration = noteDurationMap[this.data.noteDuration] || 0.25;
                    const secondsPerBeat = 60.0 / bpm;
                    const stepDuration = secondsPerBeat * 4 * baseNoteDuration * (1 / representativeEvent.speed) * representativeEvent.duration;
                    const midiNotesToBroadcast = [];

                    if (Array.isArray(event)) {
                        for (const note of event) {
                            if (note.midi !== null) {
                                this.scheduleNote(note.midi, stepDuration * note.elongation, this.nextStepTime);
                                midiNotesToBroadcast.push(note.midi);
                            }
                        }
                    } else {
                        if (event.midi !== null) {
                             this.scheduleNote(event.midi, stepDuration * event.elongation, this.nextStepTime);
                             midiNotesToBroadcast.push(event.midi);
                        }
                    }

                    if (this.midiListeners.length > 0 && midiNotesToBroadcast.length > 0) {
                        this.midiListeners.forEach(callback => callback(midiNotesToBroadcast));
                    }

                    this.nextStepTime += stepDuration;
                    this.currentStep = (this.currentStep + 1) % this.sequenceEvents.length;
                }

                this.timeoutId = setTimeout(() => this.schedule(), 25);
            }

            stop() {
                if (!this.started) return;
                clearTimeout(this.timeoutId);
                this.timeoutId = null;
                this.started = false;
                console.log('Sequencer stopped.');
            }

            // Method to update parameters dynamically (called from Rete.js control changes)
            updateParameter(param, value) {
                this.data[param] = value;
                if (this.started) {
                    // Restart sequencer to apply new tempo/notes
                    this.stop();
                    this.start();
                }
            }

            generateRandomSequence() {
                const randomizeMode = this.data.randomizeMode || 'Chord';
                const chordType = this.data.chordType || 'Major Triad';
                const scaleType = this.data.scaleType || 'Major (Ionian)';
                const rootNote = this.data.rootNote || 'C4';
                const octaveRange = this.data.octaveRange !== undefined ? this.data.octaveRange : 0;
                const pattern = this.data.arpeggioPattern || 'Random';

                const intervals = randomizeMode === 'Chord'
                    ? (CHORD_TYPES[chordType] || CHORD_TYPES['Major Triad'])
                    : (SCALE_TYPES[scaleType] || SCALE_TYPES['Major (Ionian)']);

                const rootMidi = ROOT_NOTES[rootNote] || ROOT_NOTES['C4'];

                // Reverse map for MIDI to Note Name
                const midiToNoteName = {};
                for (const [name, midi] of Object.entries(ROOT_NOTES)) {
                    midiToNoteName[midi] = name;
                }

                let allNotesMidi = [];
                for (let oct = 0; oct <= octaveRange; oct++) {
                    const offset = oct * 12;
                    intervals.forEach(interval => {
                        const midi = rootMidi + interval + offset;
                        allNotesMidi.push(midi);
                    });
                }
                allNotesMidi.sort((a, b) => a - b);

                let selectedNotesMidi = [];
                switch (pattern) {
                    case 'Down':
                        selectedNotesMidi = [...allNotesMidi].reverse();
                        break;
                    case 'Up-Down':
                        selectedNotesMidi = allNotesMidi.concat([...allNotesMidi].reverse().slice(1, -1));
                        break;
                    case 'Down-Up':
                        selectedNotesMidi = [...allNotesMidi].reverse().concat(allNotesMidi.slice(1, -1));
                        break;
                    case 'Random':
                        selectedNotesMidi = [];
                        for (let i = 0; i < 16; i++) {
                            if (Math.random() < 0.2) {
                                selectedNotesMidi.push(null); // represent rest
                            } else {
                                selectedNotesMidi.push(allNotesMidi[Math.floor(Math.random() * allNotesMidi.length)]);
                            }
                        }
